#include <iostream>
#include <map>

map<string, string> codonTable = {
  {"AUG", "Methionine"}, {"UUU", "Phenylalanine"},
  {"UAA", "STOP"}, {"UAG", "STOP"}, {"UGA", "STOP"}
};
string getAA(string codon) {
  return codonTable.count(codon) ? codonTable[codon] : "Invalid";
}
